import math

class Robot:
  def __init__(self, loc, theta ):
    self.updatePosition( loc )
    self.theta = theta

  def forward( self, dist, stddev = 0 ):
    x,y = self.loc
    if stddev > 0:
      dist = random.gauss( dist, stddev )

    x = x + math.cos(self.theta) * dist
    y = y + math.sin(self.theta) * dist
    self.loc = [x,y]

  def turn(self, dTheta, stddev = 0 ):
    if stddev > 0:
      dTheta = random.gauss( dTheta, stddev )
    self.theta = Robot.normalizeAngle( self.theta + dTheta )

  def updatePosition( self, loc ):
    self.loc = loc

  @staticmethod
  def normalizeAngle( angle ):
    while angle >= 2.0 * math.pi:
      angle = angle - 2.0 * math.pi
    while angle < 0:
      angle = angle + 2.0 * math.pi
    return angle
